policy_rule {
  name: "ssd_only_volume"
  description: "Place files on SSD for specific volume"
  enabled: true
  scope {
    volume {
      name_regex: "ssd_volume"
    }
  }
  policies {
    file_tag_based_placement {
      required_tag: "ssd"
      soften_tag_requirements: false
      automatic_constraints: NONE
    }
  }
}
